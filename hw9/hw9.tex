\begin{enumerate}
    \item 添加以下语义规则：
    
    \begin{align*}
        S_1.next &= newlabel() \\
        B.true &= S.next \\
        B.false &= newlabel() \\
        S.code &= label(B.false) \| S_1.code \| label(S_1.next) \| B.code 
    \end{align*}

    \item 添加以下语义规则：
    
    \begin{align*}
        S_1.next &= newlabel() \\
        B.true &= newlabel() \\
        B.false &= S.next \\
        S_2.next &= S_1.next \\
        S_3.next &= newlabel() \\
        S.code &= S_1.code \| label(S_1.next) \| B.code \| label(B.true) \\
        &\,\, \| \,\, S_3.code \| label(S_3.next) \| S_2.code \| gen('goto', S_1.next)
    \end{align*}
\end{enumerate}

\begin{enumerate}
    \item 翻译过程如下：
    
    \begin{center}
        \begin{tabular}{|c|c|c|l|}
            \hline
            子表达式 & $truelist$ & $falselist$ & 翻译结果 \\
            \hline
            $E_1 = a==b$ & \{100\} & \{101\} & 100: \texttt{if a==b goto _} \\
            & & & 101: \texttt{goto _} \\
            \hline
            $E_2 = c==d$ & \{102\} & \{103\} & 102: \texttt{if c==d goto _} \\
            & & & 103: \texttt{goto _} \\
            \hline
            $E_3 = e==f$ & \{104\} & \{105\} & 104: \texttt{if e==f goto _} \\
            & & & 105: \texttt{goto _} \\
            \hline
            $E_4 = E_2 \| E_3$ & $E_2.t \cup E_3.t = \{102, 104\}$ & $E_3.f = \{105\}$ & 回填104至103 \\
            \hline
            $E = E_1 \&\& E_4$ & $E_4.t = \{102, 104\}$ & $E_1.f \cup E_4.f = \{101, 105\}$ & 回填102至100 \\
            \hline
        \end{tabular}
    \end{center}

    最终翻译结果为：

    \begin{lstlisting}
    100: if a==b goto 102
    101: goto _
    102: if c==d goto _
    103: goto 104
    104: if e==f goto _
    105: goto _
    \end{lstlisting}
